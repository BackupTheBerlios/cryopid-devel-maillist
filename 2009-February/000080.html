fork() just returns different values, it does not fiddle with the
program's stack. This code demonstrates that the process:

#include <stdio.h>
#include <unistd.h>

int main() {
	pid_t pid;
	switch (pid = fork()) {
		case -1:
			printf("fork() error\n");
			break;
		case 0:
			printf("got %i  am %i (child)\n", pid, getpid());
			break;
		default:
			printf("got %i  am %i (parent)\n", pid, getpid());
	}
}

Example run:

got 0  am 3669 (child)
got 3669  am 3668 (parent)
